316  template<class KeyValueT, class NVTypeT, class IndexType, size_t tmp_nodes = 32>
317  class NodeAllocatorCompact {
318  protected:
319  //~ static_assert(std::is_trivially_copyable<KeyValueT>::value,
320  //~ "Key and Value must be trivially copyable to use compact flat allocator!\n");
321  typedef IndexType NodeHandle;
322  typedef KeyValueT KeyValue;
323  typedef NVTypeT NVType;
324 
325  private:
326  static constexpr IndexType redbit = 1U << (std::numeric_limits<IndexType>::digits - 1);
327  static constexpr IndexType max_nodes = redbit - 1;
328  static constexpr IndexType deleted_indicator = (max_nodes | redbit); /* 0xFFFFFFFFh */
329 
330  protected:
332  struct Node {
333  protected:
335  KeyValue kv;
337  IndexType parent;
338  IndexType left;
339  IndexType right;
340  /* red-black flag is stored inside parent */
341 
342  public:
343  Node(const KeyValue& kv_) : kv(kv_), parent(NodeAllocatorCompact::Invalid),
344  left(NodeAllocatorCompact::Invalid), right(NodeAllocatorCompact::Invalid) { }
345  Node(KeyValue&& kv_) : kv(kv_), parent(NodeAllocatorCompact::Invalid),
346  left(NodeAllocatorCompact::Invalid), right(NodeAllocatorCompact::Invalid) { }
347  template<class... T> Node(T&&... args) : kv(std::forward<T...>(args...)), parent(NodeAllocatorCompact::Invalid),
348  left(NodeAllocatorCompact::Invalid), right(NodeAllocatorCompact::Invalid) { }
349  Node():kv(), parent(NodeAllocatorCompact::Invalid),
350  left(NodeAllocatorCompact::Invalid), right(NodeAllocatorCompact::Invalid) { }
351 
352  KeyValue& get_key_value() { return kv; }
353  const KeyValue& get_key_value() const { return kv; }
354  bool is_red() const { return parent & NodeAllocatorCompact::redbit; }
355  bool is_black() const { return !is_red(); }
356  void set_red() { parent |= NodeAllocatorCompact::redbit; }
357  void set_black() { parent &= (~NodeAllocatorCompact::redbit); }
358 
359  NodeHandle get_parent() const { return parent & (~NodeAllocatorCompact::redbit); }
360  NodeHandle get_left() const { return left; }
361  NodeHandle get_right() const { return right; }
362  void set_parent(NodeHandle p) {
363  if(p > NodeAllocatorCompact::max_nodes) throw std::runtime_error("NodeAllocatorCompact::Node::set_parent(): parent ID too large!\n");
364  parent = p | (parent & NodeAllocatorCompact::redbit);
365  }
366  void set_left(NodeHandle x) { left = x; }
367  void set_right(NodeHandle x) { right = x; }
368 
370  void set_deleted() { parent = NodeAllocatorCompact::deleted_indicator; }
372  bool is_deleted() const { return (parent == NodeAllocatorCompact::deleted_indicator); }
373 
374  /* swap contents with other node */
375  void swap(Node& n) {
376  using std::swap;
377  swap(kv,n.kv);
378  swap(parent,n.parent);
379  swap(left,n.left);
380  swap(right,n.right);
381  }
382 
383  friend class NodeAllocatorCompact<KeyValueT,NVTypeT,IndexType,tmp_nodes>;
384  };

426  bool try_get_node(NodeHandle& n) {
427  if(deleted_nodes_head != Invalid) {
428  if(!n_del) throw std::runtime_error("NodeAllocatorCompact::alloc_node(): inconsistent deleted nodes!\n");
429  n_del--;
430  n = deleted_nodes_head;
431  nodes[n].set_parent(Invalid);
432  deleted_nodes_head = nodes[deleted_nodes_head].get_left();
433  if(deleted_nodes_head != Invalid) nodes[deleted_nodes_head].set_right(Invalid);
434  return true;
435  }
436  else return false;
437  }

440  void move_nv(IndexType x, IndexType y) {
441  size_t xbase = ((size_t)x)*nv_per_node;
442  size_t ybase = ((size_t)y)*nv_per_node;
443  for(unsigned int i=0;i<nv_per_node;i++) nvarray[xbase + i] = nvarray[ybase + i];
444  }
445 
451  void move_node(NodeHandle x, NodeHandle y) {
452  nodes[x] = std::move(nodes[y]); /* no need to swap (i.e. preserve values in x) */
453  move_nv(x,y); /* move rank function as well */
454  /* note: destructor is not called for y (memory is not freed) */
455  if(y == root || y == nil) throw std::runtime_error("NodeAllocatorCompact::move_node(): root or nil is moved!\n");
456  /* update parent and children if needed */
457  if(nodes[x].get_parent() != Invalid && nodes[x].get_parent() != nil) {
458  if(nodes[nodes[x].get_parent()].get_left() == y) nodes[nodes[x].get_parent()].set_left(x);
459  else {
460  if(nodes[nodes[x].get_parent()].get_right() == y) nodes[nodes[x].get_parent()].set_right(x);
461  else throw std::runtime_error("NodeAllocatorCompact::move_node(): inconsistent tree detected!\n");
462  }
463  }
464  else throw std::runtime_error("NodeAllocatorCompact::move_node(): node with no parent!\n");
465  if(nodes[x].get_left() != Invalid && nodes[x].get_left() != nil) {
466  if(nodes[nodes[x].get_left()].get_parent() != y)
467  throw std::runtime_error("NodeAllocatorCompact::move_node(): inconsistent tree detected!\n");
468  nodes[nodes[x].get_left()].set_parent(x);
469  }
470  if(nodes[x].get_right() != Invalid && nodes[x].get_right() != nil) {
471  if(nodes[nodes[x].get_right()].get_parent() != y)
472  throw std::runtime_error("NodeAllocatorCompact::move_node(): inconsistent tree detected!\n");
473  nodes[nodes[x].get_right()].set_parent(x);
474  }
475  }

576  void shrink_to_fit() {
577  if(!tmp_nodes) {
578  while(deleted_nodes_head != Invalid) {
579  /* remove nodes at the end */
580  NodeHandle size = nodes.size();
581  if(size == 0) throw std::runtime_error("NodeAllocatorCompact::shrink_size(): ran out of nodes!\n");
582  size--;
583  if(nodes[size].is_deleted()) {
584  /* if last node was deleted, update the list of deleted nodes */
585  NodeHandle left = nodes[size].get_left();
586  NodeHandle right = nodes[size].get_right();
587  if(left != Invalid) nodes[left].set_right(right);
588  if(right != Invalid) nodes[right].set_left(left);
589  if(deleted_nodes_head == size) deleted_nodes_head = left;
590  }
591  else {
592  /* otherwise move last node into the place of a deleted node */
593  NodeHandle n = deleted_nodes_head;
594  if(n == size)
595  throw std::runtime_error("NodeAllocatorCompact::shrink_size(): inconsistent deleted nodes!\n");
596  deleted_nodes_head = nodes[deleted_nodes_head].get_left();
597  nodes[deleted_nodes_head].set_right(Invalid);
598  move_node(n,size);
599  }
600  nodes.pop_back();
601  nvarray.resize(((size_t)size)*nv_per_node);
602  if(!n_del) throw std::runtime_error("NodeAllocatorCompact::shrink_size(): inconsistent deleted nodes!\n");
603  n_del--;
604  }
605  shrink_memory();
606  }
607  }
