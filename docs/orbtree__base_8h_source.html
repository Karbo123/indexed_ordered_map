314  template<class NodeAllocator, class Compare, class NVFunc, bool multi> template<class K>
315  auto orbtree_base<NodeAllocator,Compare,NVFunc,multi>::find(const K& key) const -> NodeHandle {
316  if(root() == Invalid) return nil();
317  NodeHandle n = get_node(root()).get_right();
318  if(n == Invalid || n == nil()) return nil();
319  while(true) {
320  const KeyType& k1 = get_node_key(n);
321  if(c(key,k1)) {
322  /* key < k1 */
323  n = get_node(n).get_left();
324  }
325  else {
326  if(c(k1,key)) n = get_node(n).get_right(); /* key > k1 */
327  else return n; /* key == k1 */
328  }
329  if(n == nil()) return n; /* end of search, not found */
330  }
331  }

333  template<class NodeAllocator, class Compare, class NVFunc, bool multi> template<class K>
334  auto orbtree_base<NodeAllocator,Compare,NVFunc,multi>::lower_bound(const K& key) const -> NodeHandle {
335  if(root() == Invalid) return nil();
336  NodeHandle n = get_node(root()).get_right();
337  if(n == Invalid || n == nil()) return nil();
338  NodeHandle last = nil(); /* guess of the result node */
339  while(true) {
340  const KeyType& k1 = get_node_key(n);
341  if(c(k1,key)) {
342  /* key > k1 */
343  n = get_node(n).get_right();
344  }
345  else { /* key <= k1, potential candidate */
346  last = n;
347  n = get_node(n).get_left();
348  }
349  if(n == nil()) return last; /* end of search */
350  }
351  }

353  template<class NodeAllocator, class Compare, class NVFunc, bool multi> template<class K>
354  auto orbtree_base<NodeAllocator,Compare,NVFunc,multi>::upper_bound(const K& key) const -> NodeHandle {
355  if(root() == Invalid) return nil();
356  NodeHandle n = get_node(root()).get_right();
357  if(n == Invalid || n == nil()) return nil();
358  NodeHandle last = nil(); /* guess of the result node */
359  while(true) {
360  const KeyType& k1 = get_node_key(n);
361  if(c(key,k1)) {
362  /* key < k1, potential candidate */
363  last = n;
364  n = get_node(n).left;
365  }
366  else {
367  /* key >= k1, has to go right */
368  n = get_node(n).right;
369  }
370  if(n == nil()) return last; /* end of search */
371  }

374  template<class NodeAllocator, class Compare, class NVFunc, bool multi> template<class K>
375  void orbtree_base<NodeAllocator,Compare,NVFunc,multi>::get_sum_fv(const K& k, NVType* res) const {
376  for(unsigned int i=0; i < f.get_nr(); i++) res[i] = NVType();
377  NVType tmp[f.get_nr()];
378  if(root() == Invalid) return;
379  //~ const Node& r = get_node(root());
380  NodeHandle n = get_node(root()).get_right();
381  if(n == Invalid || n == nil()) return;
382  while(true) { /* recursive search starting from the root to find all nodes with key < k */
383  const KeyType& k1 = get_node_key(n);
384  if(c(k1,k)) {
385  /* k1 < key, we have to add the sum from the left subtree + n and continue to the right */
386  NodeHandle l = get_node(n).get_left();
387  if(l != nil()) {
388  this->get_node_sum(l,tmp);
389  NVAdd(res,tmp);
390  }
391  get_node_grvalue(n,tmp);
392  NVAdd(res,tmp);
393 
394  n = get_node(n).get_right();
395  if(n == nil()) break;
396  }
397  else {
398  /* k1 >= key, we have to continue toward the left */
399  n = get_node(n).get_left();
400  if(n == nil()) break;
401  }
402  }

405  template<class NodeAllocator, class Compare, class NVFunc, bool multi>
406  void orbtree_base<NodeAllocator,Compare,NVFunc,multi>::get_sum_fv_node(NodeHandle x, NVType* res) const {
407  for(unsigned int i=0; i < f.get_nr(); i++) res[i] = NVType();
408  NVType tmp[f.get_nr()];
409  if(x == this->Invalid || x == nil() || x == root()) return;
410  /* 1. add sum from x's left to the sum
411  * 2. go up one level
412  * 3. if x is right child, add parent's value + parent's left child's value
413  * 4. repeat until root is reached */
414  NodeHandle l = get_node(x).get_left();
415  if(l != nil()) {
416  this->get_node_sum(l,tmp);
417  NVAdd(res,tmp);
418  }
419 
420  NodeHandle p = get_node(x).get_parent();
421 
422  while(p != root()) {
423  if(x == get_node(p).get_right()) {
424  l = get_node(p).get_left();
425  if(l != nil()) {
426  this->get_node_sum(l,tmp);
427  NVAdd(res,tmp);
428  }
429  get_node_grvalue(p,tmp);
430  NVAdd(res,tmp);
431  }
432  x = p;
433  p = get_node(x).get_parent();
434  }
435  }

453  auto orbtree_base<NodeAllocator,Compare,NVFunc,multi>::first() const -> NodeHandle {
454  if(root() == Invalid) return nil();
455  NodeHandle n = get_node(root()).get_right();
456  if(n == Invalid || n == nil()) return nil();
457  /* keep going left until it's possible */
458  while(get_node(n).get_left() != nil()) n = get_node(n).get_left();

461  template<class NodeAllocator, class Compare, class NVFunc, bool multi>
462  auto orbtree_base<NodeAllocator,Compare,NVFunc,multi>::last() const -> NodeHandle {
463  if(root() == Invalid) return nil();
464  NodeHandle n = get_node(root()).get_right();
465  if(n == Invalid || n == nil()) return nil();
466  /* keep going right until it's possible */
467  while(get_node(n).get_right() != nil()) n = get_node(n).get_right();
468  return n;
469  }
470  template<class NodeAllocator, class Compare, class NVFunc, bool multi>
471  auto orbtree_base<NodeAllocator,Compare,NVFunc,multi>::next(NodeHandle n) const -> NodeHandle {
472  if(n == nil()) return n; /* incrementing nil is no-op */
473  /* try going right */
474  if(get_node(n).get_right() != nil()) {
475  n = get_node(n).get_right();
476  /* and then keep going left until possible */
477  while(get_node(n).get_left() != nil()) n = get_node(n).get_left();
478  return n;
479  }
480  /* if not possible to go right, go up until n is a right child */
481  while(true) {
482  NodeHandle p = get_node(n).get_parent();
483  if(p == root()) return nil(); /* end of search, no next node fonud */
484  if(get_node(p).get_left() == n) return p; /* found a suitable parent */
485  n = p; /* keep going */
486  }
487  }
488  template<class NodeAllocator, class Compare, class NVFunc, bool multi>
489  auto orbtree_base<NodeAllocator,Compare,NVFunc,multi>::previous(NodeHandle n) const -> NodeHandle {
490  if(n == nil()) return last(); /* decrementing nil results in the last valid node -- this is done to make implementing the end() iterator easier */
491  /* try going left */
492  if(get_node(n).get_left() != nil()) {
493  n = get_node(n).get_left();
494  /* keep going right if possible */
495  while(get_node(n).get_right() != nil()) n = get_node(n).get_right();
496  return n;
497  }
498  /* if not possible to go left, go up until n is a left child */
499  while(true) {
500  NodeHandle p = get_node(n).get_parent();
501  if(p == root()) return nil(); /* already at the beginning */
502  if(get_node(p).get_right() == n) return p; /* found a suitable parent */
503  n = p; /* otherwise keep going */
504  }
505  }

511  template<class NodeAllocator, class Compare, class NVFunc, bool multi>
512  void orbtree_base<NodeAllocator,Compare,NVFunc,multi>::update_sum(NodeHandle n) {
513  /* sum(n) = sum(n.left) + sum(n.right) + f(n.key) */
514  NVType sum[f.get_nr()];
515  NVType tmp[f.get_nr()];
516  get_node_grvalue(n,sum);
517  if(get_node(n).get_left() != nil()) {
518  this->get_node_sum(get_node(n).get_left(),tmp);
519  NVAdd(sum,tmp);
520  }
521  if(get_node(n).get_right() != nil()) {
522  this->get_node_sum(get_node(n).get_right(),tmp);
523  NVAdd(sum,tmp);
524  }
525  this->set_node_sum(n,sum);
526  }

529  /* left rotate:
530  * right child of x takes its place, x becomes its left child
531  * left child of x's original right child becomes x's right child
532  */
533  template<class NodeAllocator, class Compare, class NVFunc, bool multi>
534  void orbtree_base<NodeAllocator,Compare,NVFunc,multi>::rotate_left(NodeHandle x) {
535  NodeHandle y = get_node(x).get_right(); /* we assume that x != nil and y != nil */
536  get_node(x).set_right( get_node(y).get_left() ); /* x.right = y.left; -- this can be nil */
537  if(get_node(x).get_right() != nil()) get_right(x).set_parent(x); /* here we have to check -- don't want to modify parent of nil */
538  get_node(y).set_parent( get_node(x).get_parent() );
539 
540  if( x == get_parent(x).get_right() ) get_parent(x).set_right(y); /* note: parent of x can be root (sentinel) node (if x is the "real" root of the tree */
541  else get_parent(x).set_left(y);
542  get_node(y).set_left(x);
543  get_node(x).set_parent(y);
544 
545  update_sum(x); /* only these two need to be updated -- upstream of y the values stay the same */
546  update_sum(y);
547  }

549  /* right rotate.
550  * left child of x takes its place, x becomes its right child
551  * right child of x's original left child becomes x's left child */
552  template<class NodeAllocator, class Compare, class NVFunc, bool multi>
553  void orbtree_base<NodeAllocator,Compare,NVFunc,multi>::rotate_right(NodeHandle x) {
554  NodeHandle y = get_node(x).get_left(); /* we assume that x != nil and y != nil */
555  get_node(x).set_left( get_node(y).get_right() );
556  if(get_node(x).get_left() != nil()) get_left(x).set_parent(x);
557  get_node(y).set_parent( get_node(x).get_parent() );
558 
559  if( x == get_parent(x).get_right() ) get_parent(x).set_right(y);
560  else get_parent(x).set_left(y);
561  get_node(y).set_right(x);
562  get_node(x).set_parent(y);
563 
564  update_sum(x);
565  update_sum(y);
566  }

577  /* helper for insert -- find the location where to insert
578  * note: in a multi map/set, the inserted element will come after any already
579  * existing elements with the same key */
580  template<class NodeAllocator, class Compare, class NVFunc, bool multi>
581  bool orbtree_base<NodeAllocator,Compare,NVFunc,multi>::insert_search(const KeyType& k, NodeHandle& n, bool& insert_left) const {
582  n = root();
583  if(n == Invalid) throw std::runtime_error("orbtree_base::insert_search(): root is Invalid!\n");
584  if(get_node(n).get_right() == nil()) {
585  /* empty tree, insert as the dummy root node's right child */
586  insert_left = false;
587  return true;
588  }
589  n = get_node(n).get_right(); /* nonempty tree, start with the real root */
590  while(true) {
591  const KeyType& k1 = get_node_key(n);
592  if(c(k,k1)) {
593  /* should go to the left */
594  if(get_node(n).get_left() == nil()) { insert_left = true; return true; }
595  n = get_node(n).get_left();
596  }
597  else {
598  /* should go to the right */
599  /* first check if key exists (if not multimap / multiset) */
600  if(!multi) if(!c(k1,k)) return false;
601  if(get_node(n).get_right() == nil()) { insert_left = false; return true; }
602  n = get_node(n).get_right();
603  }
604  }
605  }

608  /* helper function to do the real work for insert -- insert n1 as the left / right child of n
609  * n1 must already contain the proper key / value, but can be uninitialized otherwise
610  * note: this function does not check for the correct relationship between the keys of n and n1,
611  * that is the caller's responsibility */
612  template<class NodeAllocator, class Compare, class NVFunc, bool multi>
613  void orbtree_base<NodeAllocator,Compare,NVFunc,multi>::insert_helper(NodeHandle n, NodeHandle n1, bool insert_left) {
614  if(insert_left) get_node(n).set_left(n1);
615  else get_node(n).set_right(n1);
616  get_node(n1).set_parent(n);
617  get_node(n1).set_left(nil());
618  get_node(n1).set_right(nil());
619  get_node(n1).set_red(); /* all new nodes are red */
620  NVType sum_add[f.get_nr()];
621  get_node_grvalue(n1,sum_add); /* calculate the new value */
622  this->set_node_sum(n1,sum_add);
623  /* update sum up the tree from n */
624  for(NodeHandle n2 = n; n2 != root(); n2 = get_node(n2).get_parent()) {
625  NVType tmp[f.get_nr()];
626  this->get_node_sum(n2,tmp);
627  NVAdd(tmp,sum_add);
628  this->set_node_sum(n2,tmp);
629  }
630 
631  while(true) {
632  if(n == root()) return; /* nothing to do if we just added one node to an empty tree */
633  /* here, n1 is always red */
634  /* has to fix red-black tree properties */
635  /* only possible problem is if n is red */
636  if(get_node(n).is_black()) return;
637 
638  /* if n is red, it can have a black other child (only after the first step though) */
639 
640  /* if n is the real root, we can fix the tree by coloring it black */
641  if(get_node(n).get_parent() == root()) { get_node(n).set_black(); return; }
642 
643  /* now, n has a valid, black parent and potentially a sibling */
644  if(get_sibling(n).is_red()) {
645  /* easier case, we can color n and its sibling black and n's parent red
646  * note: nil is always black, so this will only be true if n has a real sibling */
647  get_sibling(n).set_black();
648  get_node(n).set_black();
649  get_parent(n).set_red();
650  /* iteratoion has to continue to check if n's grandparent was red or black */
651  n1 = get_node(n).get_parent();
652  n = get_node(n1).get_parent();
653  }
654  else {
655  /* n has either a black sibling or no sibling -- here, we have to do rotations,
656  * which depends on whether n and n1 are on the same side */
657  if(is_left_side(n1) != is_left_side(n)) {
658  /* different side, we have to rotate around n first */
659  rotate_parent(n1);
660  NodeHandle tmp = n1;
661  n1 = n;
662  n = tmp;
663  }
664  /* now n and n1 are on the same side, we have to rotate around n's parent */
665  get_node(n).set_black();
666  get_parent(n).set_red();
667  rotate_parent(n);
668  return; /* in this case, we are done, the "top" node (in n's parent's place) is now black */

700  template<class NodeAllocator, class Compare, class NVFunc, bool multi>
701  bool orbtree_base<NodeAllocator,Compare,NVFunc,multi>::insert_search_hint(NodeHandle hint, const KeyType& k, NodeHandle& n, bool& insert_left) const {
702  if(c(k,get_node_key(hint))) {
703  /* k should go before hint, it might be a valid hint
704  /* have to compare with previous value as well */
705  NodeHandle p = previous(hint);
706  if(!c(k,get_node_key(p))) {
707  /* p <= k < hint, might be good */
708  if(!multi) if(!c(get_node_key(p),k)) return p; /* p == k, cannot insert */
709  /* insert between p and hint
710  * note: either p does not have a right child or hint does not have a left child */
711  if(get_node(hint).get_left() == nil()) { n = hint; insert_left = true; return true; }
712  else {
713  if(get_node(p).get_right() == nil()) { n = p; insert_left = false; return true; }
714  else throw std::runtime_error("orbtree_base::insert(): inconsistent tree detected!\n");
715  }
716  }
717  /* here, hint is not good, but new element should go before it
718  * in this case, normal insert works in all cases (for multi,
719  * new element should be inserted after all elements with the same key) */
720  return insert_search(k,n,insert_left).first;
721  }
722  /* here, either hint is not good, or k == hint */
723  if(!c(get_node_key(hint),k)) {
724  if(!multi) { n = hint; return false; } /* hint == k, cannot insert */
725  if(get_node(hint).get_left() == nil()) { n = hint; insert_left = true; return true; }
726  else {
727  NodeHandle p = previous(hint);
728  if(get_node(p).get_right() != nil()) std::runtime_error("orbtree_base::insert(): inconsistent tree detected!\n");
729  n = p;
730  insert_left = false;
731  return true;
732  }
733  }
734  /* new element should go after hint, should be the first if elements with the same key already exist */
735  n = lower_bound(k);
736  if(n != nil()) { /* insert before n */
737  if(get_node(n).get_left() == nil()) { insert_left = true; return true; }
738  else {
739  NodeHandle p = previous(n);
740  if(get_node(p).get_right() != nil()) std::runtime_error("orbtree_base::insert(): inconsistent tree detected!\n");
741  n = p;
742  insert_left = false;
743  return true;
744  }
745  }
746  else {
747  n = last(); /* note: this is inefficient, as this case will require going through all levels twice (last() is O(logN) ) */
748  insert_left = false;
749  return true;
750  }
751  }

814  template<class NodeAllocator, class Compare, class NVFunc, bool multi>
815  auto orbtree_base<NodeAllocator,Compare,NVFunc,multi>::erase(NodeHandle n) -> NodeHandle {
816  /* delete node n, return a handle to its successor in the tree */
817  NodeHandle x = next(n);
818  /* if n has one child, we can delete n, replace it with one of its children
819  /* if n has two children, then x has one child (we go right from n and keep going left as long as it's possible)
820  * in this case, we can cut out x, and then move it in place of n */
821  NodeHandle del = n;
822  if(get_node(n).get_left() != nil() && get_node(n).get_right() != nil()) del = x;
823 
824  /* delete del, replace it with its only child */
825  NodeHandle c = get_node(del).get_left();
826  if(c == nil()) c = get_node(del).get_right();
827  NodeHandle p = get_node(del).get_parent();
828  get_node(c).set_parent( p ); /* c can be nil here, its parent will be set anyway */
829  if(get_node(p).get_left() == del) get_node(p).set_left(c);
830  else get_node(p).set_right(c);
831 
832  /* subtract the rank function values up from del */
833  NVType x2[f.get_nr()];
834  get_node_grvalue(del,x2);
835  for(NodeHandle p2 = p; p2 != root(); p2 = get_node(p2).get_parent()) {
836  NVType y[f.get_nr()];
837  this->get_node_sum(p2,y);
838  NVSubtract(y,x2);
839  this->set_node_sum(p2,y);
840  }
841 
842  /* cut out del, we need to fix the tree properties */
843  /* cases:
844  * 1. del is black, c is red -> color c black
845  * 2. del is red, c is black -> no need to do anything
846  * 3. del is black, c is nil -> one less black on del's path, need to fix it
847  */
848  if(get_node(del).is_black()) {
849  if(c != nil()) {
850  get_node(c).set_black();
851  }
852  else {
853  /* here, del is black -- in this case, it must have a valid sibling */
854  while(true) {
855  NodeHandle s = get_node(p).get_left();
856  if(s == nil() || s == c) s = get_node(p).get_right();
857  if(s == nil() || s == c) throw std::runtime_error("orbtree_base::erase(): found black node with no sibling!\n");
858 
859  if(get_node(s).is_red()) {
860  /* p is black */
861  /* recolor p as red, s as black, do a rotation on p and continue from there */
862  get_node(p).set_red();
863  get_node(s).set_black();
864  rotate_parent(s);
865  continue; /* continue with the same p, which is now red and its other child is black */
866  }
867  /* here, s is black, p can be black or red */
868  if(get_left(s).is_black() && get_right(s).is_black()) {
869  /* two black children (or nil) */
870  /* we can color s red and fix things on upper level */
871  get_node(s).set_red();
872  /* if parent is red, we can fix the tree by coloring it black and s red */
873  if(get_node(p).is_red()) {
874  get_node(p).set_black();
875  break;
876  }
877  /* if p was black, we have to continue one level up */
878  c = p;
879  p = get_node(p).get_parent();
880  if(p == root()) break;
881  continue;
882  }
883  /* here, s is black and at least one child of it is red
884  * if it is on the other side as s, then we need to rotate s,
885  * so that the red child is on the same side */
886  if(s == get_node(p).get_right() && get_right(s).is_black()) {
887  /* s is right child, but left child of s is red, rotate right */
888  get_node(s).set_red();
889  get_left(s).set_black();
890  rotate_right(s);
891  continue; /* re-assign s at the top of the loop -- p did not change */
892  }
893  /* previous condition reversed */
894  if(s == get_node(p).get_left() && get_left(s).is_black()) {
895  /* s is left child, but right child of s is red, rotate left */
896  get_node(s).set_red();
897  get_right(s).set_black();
898  rotate_left(s);
899  continue;
900  }
901 
902  /* here s is black and has a red child on the same side as s
903  * in this case, the tree can be fixed by rotating s into p's position */
904  if(get_node(p).is_red()) get_node(s).set_red();
905  get_node(p).set_black();
906  if(s == get_node(p).get_right()) get_right(s).set_black();
907  else get_left(s).set_black();
908  rotate_parent(s);
909  break;
910  }
911 
912 
913  } /* else (c is nil) */
914  } /* if del is black -- need to fix the tree */
915 
916  if(del != n) {
917  /* x (n's successor) was deleted, have to put x in n's place
918  * in this case, x cannot be nil */
919  get_node(x).set_left( get_node(n).get_left() );
920  get_node(x).set_right( get_node(n).get_right() );
921  get_node(x).set_parent( get_node(n).get_parent() );
922  if(get_node(n).is_black()) get_node(x).set_black();
923  else get_node(x).set_red();
924  if(get_parent(n).get_left() == n) get_parent(n).set_left(x);
925  else {
926  if(get_parent(n).get_right() == n) get_parent(n).set_right(x);
927  else throw std::runtime_error("orbtree_base::erase(): inconsistent tree detected!\n");
928  }
929  if(get_node(x).get_left() != nil()) get_left(x).set_parent(x);
930  if(get_node(x).get_right() != nil()) get_right(x).set_parent(x);
931 
932  /* fix sums */
933  update_sum_r(x);
934  }
935  /* delete node of n -- need to give x as well as its value can potentially change */
936  this->free_node(n);
937  if(!size1) throw std::runtime_error("size1 is zero in orbtree_base::erase!\n");
938  size1--;
939  return x; /* return the successor -- it can be nil if n was the largest node */
940  }

991  void orbtree_base<NodeAllocator,Compare,NVFunc,multi>::check_tree(double epsilon) const {

993  const Node& r = get_node(root());
994  if(r.get_left() != nil() || r.get_parent() != nil()) throw std::runtime_error("orbtree_base::check_tree(): root is invalid!\n");
995  NodeHandle x = r.get_right();
996  if(x == nil()) return; /* empty tree nothing more to do */
997  if(x == this->Invalid) throw std::runtime_error("orbtree_base::check_tree(): invalid node handle found!\n");
998  if(get_node(x).get_parent() != root()) throw std::runtime_error("orbtree_base::check_tree(): inconsistent root node!\n");
999

1001  check_tree_r(epsilon,x,0,previous_black_count);
1002 
1003  }
1004 
1005  /* recursive helper function to check tree
1006  * this function checks:
1007  * -- if node x has any children, then they are consistent (i.e. have x as parent, keys are ordered properly)
1008  * -- for red x, both children have to be black or nil
1009  * -- if nil is reached, black_count has to be the same as previous_black_count
1010  * -- if epsilon >= 0.0, then than the rank function value stored in x is equal to the sum of its children + x's value
1011  * -- recurses into both children, increasing black_count if x is black
1012  * as tree height for N elements is maximum 2*log_2(N), this will not result in stack overflow
1013  * (e.g. N <~ 2^40 on a machine with few hundred GB RAM, thus tree height <~ 80, which is reasonable for recursion depth)
1014  */
1015  template<class NodeAllocator, class Compare, class NVFunc, bool multi>
1016  void orbtree_base<NodeAllocator,Compare,NVFunc,multi>::check_tree_r(double epsilon, NodeHandle x, size_t black_count, size_t& previous_black_count) const {
1017  NodeHandle l = get_node(x).get_left();
1018  NodeHandle r = get_node(x).get_right();
1019 
1020  { /* scope for sum and tmp -- no need to keep them over the recursion */
1021  NVType sum[f.get_nr()];
1022  NVType tmp[f.get_nr()];
1023 
1024  if(epsilon >= 0.0) get_node_grvalue(x,sum);
1025 
1026  if(l != nil()) {
1027  /* check if x is l's parent */
1028  if(get_node(l).get_parent() != x) throw std::runtime_error("orbtree_base::check_tree(): inconsistent node!\n");
1029  /* check that red node does not have red child */
1030  if(get_node(x).is_red() && get_node(l).is_red()) throw std::runtime_error("orbtree_base::check_tree(): inconsistent node (red parent with red child)!\n");
1031  /* check that l's key is strictly smaller than x's */
1032  if( ! c(get_node_key(l),get_node_key(x)) ) {
1033  if( !multi || c(get_node_key(x),get_node_key(l)) ) throw std::runtime_error("orbtree_base::check_tree(): inconsistent ordering!\n");
1034  }
1035  /* add l's partial sum to x's value */
1036  if(epsilon >= 0.0) {
1037  this->get_node_sum(l,tmp);
1038  NVAdd(sum,tmp);
1039  }
1040  }
1041 
1042  if(r != nil()) {
1043  /* check if x is r's parent */
1044  if(get_node(r).get_parent() != x) throw std::runtime_error("orbtree_base::check_tree(): inconsistent node!\n");
1045  /* check that red node does not have red child */
1046  if(get_node(x).is_red() && get_node(r).is_red()) throw std::runtime_error("orbtree_base::check_tree(): inconsistent node (red parent with red child)!\n");
1047  /* check that r's key is larger than x's or equal to it (if multiset / multimap) */
1048  if( c(get_node_key(r),get_node_key(x)) ) throw std::runtime_error("orbtree_base::check_tree(): inconsistent ordering!\n");
1049  if( !multi && ! c(get_node_key(x),get_node_key(r)) ) throw std::runtime_error("orbtree_base::check_tree(): non-unique key found!\n");
1050 
1051  /* add r's partial sum to x's value */
1052  if(epsilon >= 0.0) {
1053  this->get_node_sum(r,tmp);
1054  NVAdd(sum,tmp);
1055  }
1056  }
1057 
1058  if(epsilon >= 0.0) {
1059  /* check that the partial sum stored in x is consistent */
1060  this->get_node_sum(x,tmp);
1061 
1062  /* if NVType is integral, we want exact match -- otherwise, we use epsilon for comparison */
1063  if(std::is_integral<NVType>::value) for(unsigned int i=0;i<f.get_nr();i++) if(tmp[i] != sum[i])
1064  throw std::runtime_error("orbtree_base::check_tree(): partial sums are inconsistent!\n");
1065  else for(unsigned int i=0;i<f.get_nr();i++) if(fabs(tmp[i]-sum[i]) > epsilon)
1066  throw std::runtime_error("orbtree_base::check_tree(): partial sums are inconsistent!\n");
1067  }
1068  }
1069 
1070  /* increase black count for recursion */
1071  if(get_node(x).is_black()) black_count++;
1072  if(r == nil() || l == nil()) {
1073  /* l or r is nil, check if black_count is same as previously */
1074  if(previous_black_count == (size_t)-1) previous_black_count = black_count;
1075  else if(previous_black_count != black_count) throw std::runtime_error("orbtree_base::check_tree(): inconsistent node (black conut differs)!\n");
1076  }
1077 
1078  /* recurse into both children (if not nil) */
1079  if(l != nil()) check_tree_r(epsilon,l,black_count,previous_black_count);
1080  if(r != nil()) check_tree_r(epsilon,r,black_count,previous_black_count);
1081  }
